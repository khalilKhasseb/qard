# Laravel Scheduler Cron Job Setup
# Add this to your server's crontab: crontab -e

# QCard Laravel Scheduler - runs every minute
* * * * * cd /var/www/qcard && php artisan schedule:run >> /dev/null 2>&1

# Alternative with logging (for debugging)
# * * * * * cd /var/www/qcard && php artisan schedule:run >> /var/www/qcard/storage/logs/scheduler.log 2>&1

# Queue job monitoring (optional - runs every 5 minutes)
# */5 * * * * cd /var/www/qcard && php artisan queue:monitor default --max=100

# Failed job retry (optional - runs every hour)
# 0 * * * * cd /var/www/qcard && php artisan queue:retry all
